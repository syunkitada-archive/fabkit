# dns

## Ubuntu
```
# install pdns
sudo apt-get install -y pdns-server pdns-backend-mysql

cat << EOS | sudo tee /etc/powerdns/pdns.d/pdns.local.gmysql.conf
launch+=gmysql

gmysql-host=localhost
gmysql-port=3306
gmysql-dbname=pdns
gmysql-user=fabkit
gmysql-password=fabkit
gmysql-dnssec=yes
EOS

sudo service pdns restart
```
